# 实时讨论页面设计规范文档

## 1. 页面概述

### 1.1 页面职责
实时讨论页面是AI Agent市场的核心功能，负责：
- 展示AI Agent们对创意的实时分析和讨论
- 管理用户的3次发言机会和时机
- 实时显示竞价状态和Agent行为
- 提供沉浸式的多人讨论体验

### 1.2 目标用户
- **创意提交者**：观察自己创意的讨论过程
- **观察者**：学习和娱乐，体验AI讨论
- **移动用户**：通过标签切换查看不同信息

### 1.3 关键指标
- 平均观看时长 > 8分钟
- 用户发言参与率 > 60%
- 讨论完成率 > 90%
- 用户满意度 > 4.3/5

```tsx
// pages/discussion/[ideaId].tsx

import { useState, useEffect } from 'react';
import { useRouter } from 'next/router';
import { DiscussionLayout } from '@/components/discussion/DiscussionLayout';

import { useWebSocket } from '@/hooks/useWebSocket';

interface DiscussionPageProps {
  idea: IdeaDetails;
  agents: Agent[];
  initialMessages: DiscussionMessage[];
  userPermissions: UserPermissions;
}

export default function DiscussionPage({
  idea,
  agents,
  initialMessages,
  userPermissions
}: DiscussionPageProps) {
  const router = useRouter();
  const { user } = useAuth();
  const [discussionState, setDiscussionState] = useState<DiscussionState>({
    phase: 'waiting',
    timeRemaining: 600, // 10分钟
    messages: initialMessages,
    agentStates: initializeAgentStates(agents),
    currentBids: [],
    userSpeakingSlots: initializeSpeakingSlots()
  });

  // WebSocket连接管理
  const { socket, isConnected } = useWebSocket(`/discussions/${idea.id}`, {
    onMessage: handleWebSocketMessage,
    onAgentAction: handleAgentAction,
    onStateChange: handleStateChange,
    onUserJoin: handleUserJoin,
    onUserLeave: handleUserLeave
  });

  // 处理WebSocket消息
  const handleWebSocketMessage = useCallback((message: DiscussionMessage) => {
    setDiscussionState(prev => ({
      ...prev,
      messages: [...prev.messages, message]
    }));

    // 触发消息相关的UI更新
    if (message.type === 'agent_response') {
      updateAgentState(message.agentId, { lastActivity: new Date() });
    }
  }, []);

  // 处理用户发言
  const handleUserSpeak = useCallback(async (content: string, slotId: number) => {
    try {
      const response = await fetch(`/api/discussions/${idea.id}/speak`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ content, slotId })
      });

      if (response.ok) {
        // 标记发言机会已使用
        updateSpeakingSlot(slotId, true);
        
        // 发送WebSocket消息
        socket?.emit('user_speak', { content, slotId });
      }
    } catch (error) {
      console.error('Failed to submit speech:', error);
    }
  }, [idea.id, socket]);

  // 页面卸载时清理连接
  useEffect(() => {
    return () => {
      socket?.disconnect();
    };
  }, [socket]);

  return (
    <DiscussionLayout
      className={`discussion-page ${discussionState.phase}`}
      isConnected={isConnected}
    >
      {/* 讨论头部 */}
      <DiscussionHeader
        idea={idea}
        discussionState={discussionState}
        userPermissions={userPermissions}
        onLeave={() => router.push('/')}
      />

      {/* 主要内容区域 */}
      <div className="discussion-content flex-1 flex">
        {/* 桌面端三栏布局 */}
        <div className="hidden lg:flex w-full">
          {/* 左侧：Agent状态面板 */}
          <div className="w-80 flex-shrink-0">
            <AgentStatusPanel
              agents={agents}
              agentStates={discussionState.agentStates}
              currentBids={discussionState.currentBids}
              phase={discussionState.phase}
            />
          </div>

          {/* 中间：讨论主体 */}
          <div className="flex-1 flex flex-col min-w-0">
            <DiscussionStream
              messages={discussionState.messages}
              agents={agents}
              currentPhase={discussionState.phase}
              timeRemaining={discussionState.timeRemaining}
            />
            
            <UserSpeakingArea
              speakingSlots={discussionState.userSpeakingSlots}
              currentTime={600 - discussionState.timeRemaining}
              canSpeak={userPermissions.canSpeak}
              onSpeak={handleUserSpeak}
            />
          </div>

          {/* 右侧：分析面板 */}
          <div className="w-80 flex-shrink-0">
            <AnalyticsPanel
              idea={idea}
              discussionState={discussionState}
              userImpactHistory={userPermissions.impactHistory}
            />
          </div>
        </div>

        {/* 移动端标签切换布局 */}
        <div className="lg:hidden w-full">
          <MobileDiscussionTabs
            idea={idea}
            agents={agents}
            discussionState={discussionState}
            userPermissions={userPermissions}
            onUserSpeak={handleUserSpeak}
          />
        </div>
      </div>

      {/* 浮动元素 */}
      <DiscussionFloatingElements
        discussionState={discussionState}
        userPermissions={userPermissions}
      />
    </DiscussionLayout>
  );
}
```

```tsx
import { useRef, useEffect, useState } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { MessageBubble } from './MessageBubble';

export const DiscussionStream: React.FC<DiscussionStreamProps> = ({
  messages,
  agents,
  currentPhase,
  timeRemaining
}) => {
  const messagesEndRef = useRef<HTMLDivElement>(null);
  const containerRef = useRef<HTMLDivElement>(null);
  const [autoScroll, setAutoScroll] = useState(true);
  const [typingAgents, setTypingAgents] = useState<string[]>([]);

  const scrollToBottom = useCallback(() => {

    if (!containerRef.current) return;

  }, [messages, scrollToBottom]);

      {/* 阶段指示器 */}

        />
      </div>

        ref={containerRef}
        className="flex-1 overflow-y-auto p-4 space-y-4"
        onScroll={handleScroll}
      >
        {/* 讨论开始提示 */}
        {messages.length === 0 && (
          <motion.div
            initial={{ opacity: 0, y: 20 }}
            animate={{ opacity: 1, y: 0 }}

          >
            <div className="w-16 h-16 mx-auto mb-4 bg-blue-100 rounded-full flex items-center justify-center">

              AI专家们正在分析你的创意，马上就要开始激烈的讨论了！
            </p>
          </motion.div>
        )}

        {/* 消息列表 */}
        <AnimatePresence initial={false}>
          {messages.map((message, index) => (
            <motion.div
              key={message.id}
              initial={{ opacity: 0, y: 20, scale: 0.95 }}
              animate={{ opacity: 1, y: 0, scale: 1 }}
              exit={{ opacity: 0, y: -20, scale: 0.95 }}

                delay: index * 0.05 // 错开动画时间
              }}

        )}

          initial={{ opacity: 0, y: 20 }}
```

```tsx
import { useState, useCallback } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { SpeakingOpportunityCard } from './SpeakingOpportunityCard';
import { SpeechInputModal } from './SpeechInputModal';
import { ImpactPrediction } from './ImpactPrediction';

interface UserSpeakingAreaProps {
  speakingSlots: SpeakingSlot[];
  currentTime: number;
  canSpeak: boolean;
  onSpeak: (content: string, slotId: number) => Promise<void>;
}

export const UserSpeakingArea: React.FC<UserSpeakingAreaProps> = ({
  speakingSlots,
  currentTime,
  canSpeak,
  onSpeak
}) => {
  const [activeSpeechModal, setActiveSpeechModal] = useState<number | null>(null);
  const [speechContent, setSpeechContent] = useState('');
  const [isSubmitting, setIsSubmitting] = useState(false);

  // 获取当前可用的发言机会
  const availableSlots = speakingSlots.filter(slot => {
    const isInTimeWindow = currentTime >= slot.timeWindow[0] && currentTime <= slot.timeWindow[1];
    return !slot.used && isInTimeWindow;
  });

  // 获取即将到来的发言机会
  const upcomingSlots = speakingSlots.filter(slot => {
    return !slot.used && currentTime < slot.timeWindow[0];
  });

  // 处理发言提交
  const handleSubmitSpeech = useCallback(async () => {
    if (!speechContent.trim() || activeSpeechModal === null) return;

    setIsSubmitting(true);
    try {
      await onSpeak(speechContent.trim(), activeSpeechModal);
      setSpeechContent('');
      setActiveSpeechModal(null);
    } catch (error) {
      console.error('Failed to submit speech:', error);
    } finally {
      setIsSubmitting(false);
    }
  }, [speechContent, activeSpeechModal, onSpeak]);

  if (!canSpeak) {
    return (
      <div className="user-speaking-area p-4 bg-gray-100 border-t">
        <div className="text-center text-gray-600">
          <LockIcon className="w-8 h-8 mx-auto mb-2 text-gray-400" />
          <p>观看模式 - 您无法在此讨论中发言</p>
        </div>
      </div>

      {/* 发言机会概览 */}
      <div className="mb-4">
        <h3 className="text-lg font-semibold text-gray-900 mb-3">
          你的发言机会
        </h3>
        
        <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
          {speakingSlots.map((slot) => (
            <SpeakingOpportunityCard
              key={slot.id}
              slot={slot}
              currentTime={currentTime}
              isActive={availableSlots.some(s => s.id === slot.id)}
              onActivate={() => setActiveSpeechModal(slot.id)}
            />
          ))}
        </div>
      </div>

      {/* 当前可用的发言机会 */}
      {availableSlots.length > 0 && (
        <motion.div
          initial={{ opacity: 0, y: 20 }}
          animate={{ opacity: 1, y: 0 }}
          className="p-4 bg-green-50 border border-green-200 rounded-lg mb-4"
        >
          <div className="flex items-center justify-between">
            <div>
              <h4 className="font-semibold text-green-800">
                🎯 现在是发言的好时机！
              </h4>
              <p className="text-green-700 text-sm">
                {availableSlots[0].description}
              </p>
            </div>
            <button
              onClick={() => setActiveSpeechModal(availableSlots[0].id)}
              className="px-4 py-2 bg-green-500 text-white rounded-lg hover:bg-green-600 transition-colors"
            >
              立即发言
            </button>
          </div>
        </motion.div>
      )}

      {/* 即将到来的发言机会提示 */}
      {availableSlots.length === 0 && upcomingSlots.length > 0 && (
        <div className="p-4 bg-blue-50 border border-blue-200 rounded-lg">
          <div className="flex items-center space-x-3">
            <ClockIcon className="w-5 h-5 text-blue-600" />

              </p>
            </div>
          </div>
        </div>
      )}

      {/* 发言输入模态框 */}
      <AnimatePresence>
        {activeSpeechModal !== null && (
          <SpeechInputModal
            isOpen={true}
            slotInfo={speakingSlots.find(s => s.id === activeSpeechModal)!}
            content={speechContent}
            onContentChange={setSpeechContent}
            onSubmit={handleSubmitSpeech}
            onClose={() => {
              setActiveSpeechModal(null);
              setSpeechContent('');
            }}
            isSubmitting={isSubmitting}
          />
        )}
      </AnimatePresence>
    </div>
  );
};
```

## 3. 后端API规范

### 3.1 WebSocket事件处理

```typescript
// api/websocket/discussion.ts
import { Server } from 'socket.io';
import { NextApiRequest } from 'next';
import { DiscussionManager } from '@/lib/discussion/DiscussionManager';
import { AgentOrchestrator } from '@/lib/agents/AgentOrchestrator';

export interface SocketData {
  userId?: string;
  username?: string;
  discussionId?: string;
}

export interface ClientToServerEvents {
  join_discussion: (data: { discussionId: string }) => void;
  leave_discussion: (data: { discussionId: string }) => void;
  user_speak: (data: { content: string; slotId: number }) => void;
  request_agent_analysis: (data: { agentId: string; query: string }) => void;
}

export interface ServerToClientEvents {
  discussion_state: (state: DiscussionState) => void;
  new_message: (message: DiscussionMessage) => void;
  agent_activity: (activity: AgentActivity) => void;
  bid_update: (bid: Bid) => void;
  phase_change: (phase: DiscussionPhase, timeRemaining: number) => void;
  user_joined: (user: { userId: string; username: string }) => void;
  user_left: (user: { userId: string; username: string }) => void;
  error: (error: { message: string; code?: string }) => void;
}

class DiscussionSocketHandler {
  private discussionManager: DiscussionManager;
  private agentOrchestrator: AgentOrchestrator;

  constructor() {
    this.discussionManager = new DiscussionManager();
    this.agentOrchestrator = new AgentOrchestrator();
  }

  // 处理用户加入讨论
  async handleJoinDiscussion(
    socket: Socket, 
    data: { discussionId: string }
  ) {
    try {
      const { discussionId } = data;
      const userId = socket.data.userId;

      // 验证权限
      const permissions = await this.discussionManager.getUserPermissions(
        discussionId, 
        userId
      );

      if (!permissions.canView) {
        socket.emit('error', { message: 'Access denied', code: 'FORBIDDEN' });
        return;
      }

      // 加入房间
      await socket.join(discussionId);
      socket.data.discussionId = discussionId;

      // 发送当前状态
      const currentState = await this.discussionManager.getDiscussionState(discussionId);
      socket.emit('discussion_state', currentState);

      // 通知其他用户
      socket.to(discussionId).emit('user_joined', {
        userId,
        username: socket.data.username
      });

      // 记录用户活动
      await this.discussionManager.recordUserActivity(discussionId, userId, 'joined');

    } catch (error) {
      console.error('Join discussion error:', error);
      socket.emit('error', { message: 'Failed to join discussion' });
    }
  }

  // 处理用户发言
  async handleUserSpeak(
    socket: Socket,
    data: { content: string; slotId: number }
  ) {
    try {
      const userId = socket.data.userId;
      const discussionId = socket.data.discussionId;

      if (!userId || !discussionId) {
        socket.emit('error', { message: 'Invalid session' });
        return;
      }

      // 验证发言权限
      const canSpeak = await this.discussionManager.validateSpeakingPermission(
        discussionId,
        userId,
        data.slotId
      );

      if (!canSpeak.allowed) {
        socket.emit('error', { message: canSpeak.reason, code: 'FORBIDDEN' });
        return;
      }

      // 处理用户发言
      const userMessage = await this.discussionManager.processUserSpeech(
        discussionId,
        userId,
        data.content,
        data.slotId
      );

      // 广播用户消息
      this.io.to(discussionId).emit('new_message', userMessage);

      // 触发Agent响应
      await this.agentOrchestrator.processUserIntervention(
        discussionId,
        userMessage
      );

      // 标记发言机会已使用
      await this.discussionManager.markSpeakingSlotUsed(
        discussionId,
        userId,
        data.slotId
      );

    } catch (error) {
      console.error('User speak error:', error);
      socket.emit('error', { message: 'Failed to process speech' });
    }
  }

  // 处理Agent活动
  async handleAgentActivity(discussionId: string, activity: AgentActivity) {
    try {
      // 更新Agent状态
      await this.discussionManager.updateAgentState(
        discussionId,
        activity.agentId,
        activity.state
      );

      // 广播Agent活动
      this.io.to(discussionId).emit('agent_activity', activity);

      // 如果是出价活动，广播出价更新
      if (activity.type === 'bid') {
        const bid = await this.discussionManager.getBid(activity.bidId);
        this.io.to(discussionId).emit('bid_update', bid);
      }

      // 如果是消息活动，广播新消息
      if (activity.type === 'message') {
        const message = await this.discussionManager.getMessage(activity.messageId);
        this.io.to(discussionId).emit('new_message', message);
      }

    } catch (error) {
      console.error('Agent activity error:', error);
    }
  }
}

export default function handler(req: NextApiRequest, res: NextApiResponse) {
  if (res.socket.server.io) {
    res.end();

  }

    );

    );

      console.log('Socket disconnected:', socket.id);
    });
  });

  res.end();
```

```typescript
// lib/discussion/DiscussionManager.ts
export class DiscussionManager {
  private redis: Redis;
  private prisma: PrismaClient;
  private agentOrchestrator: AgentOrchestrator;

  constructor() {
    this.redis = new Redis(process.env.REDIS_URL);
    this.prisma = new PrismaClient();
    this.agentOrchestrator = new AgentOrchestrator();
  }

  // 获取讨论状态
  async getDiscussionState(ideaId: string): Promise<DiscussionState> {
    const cacheKey = `discussion:${ideaId}:state`;
    
    // 尝试从缓存获取
    const cached = await this.redis.get(cacheKey);
    if (cached) {
      return JSON.parse(cached);
    }

    // 从数据库构建状态
    const state = await this.buildDiscussionState(ideaId);
    
    // 缓存状态（30秒）
    await this.redis.setex(cacheKey, 30, JSON.stringify(state));
    
    return state;
  }

  // 处理用户发言
  async processUserSpeech(
    ideaId: string,
    userId: string,
    content: string,
    slotId: number
  ): Promise<{ messageId: string; impactPrediction: ImpactPrediction }> {
    // 创建用户消息记录
    const message = await this.prisma.discussionMessage.create({
      data: {
        ideaId,
        userId,
        content,
        type: 'user_speech',
        metadata: { slotId }
      }
    });

    // 预测影响力
    const impactPrediction = await this.predictUserImpact(
      ideaId,
      content,
      slotId
    );

    // 触发Agent响应
    await this.agentOrchestrator.processUserIntervention(ideaId, message);

    // 更新用户发言权限
    await this.markSpeakingSlotUsed(ideaId, userId, slotId);

    // 记录用户影响力
    await this.recordUserImpact(ideaId, userId, message.id, impactPrediction);

    return {
      messageId: message.id,
      impactPrediction
    };
  }

  // 验证发言权限
  async validateSpeakingPermission(
    ideaId: string,
    userId: string,
    slotId: number
  ): Promise<{ allowed: boolean; reason?: string }> {
    const discussion = await this.getDiscussionState(ideaId);
    const userSlot = discussion.userSpeakingSlots.find(s => s.id === slotId);

    if (!userSlot) {
      return { allowed: false, reason: 'Invalid speaking slot' };
    }

    if (userSlot.used) {
      return { allowed: false, reason: 'Speaking slot already used' };
    }

    const currentTime = discussion.totalTime - discussion.timeRemaining;
    const isInWindow = currentTime >= userSlot.timeWindow[0] && 
                      currentTime <= userSlot.timeWindow[1];

    if (!isInWindow) {
      return { allowed: false, reason: 'Not in speaking time window' };
    }

    return { allowed: true };
  }

  // 预测用户影响力
  private async predictUserImpact(
    ideaId: string,
    content: string,
    slotId: number
  ): Promise<ImpactPrediction> {
    try {
      const response = await fetch(`${process.env.AI_SERVICE_URL}/predict-impact`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          ideaId,
          userContent: content,
          slotId,
          discussionContext: await this.getDiscussionContext(ideaId)
        })
      });

      return await response.json();
    } catch (error) {
      console.error('Impact prediction failed:', error);
      
      // 降级处理：返回基础预测
      return {
        agreement: 0.5,
        disruption: 0.3,
        information: 0.7,
        persuasion: 0.4,
        confidence: 0.3
      };
    }
  }
}
```

```typescript
export const useWebSocket = (
  url: string,
  options: WebSocketOptions = {}

  const [isConnected, setIsConnected] = useState(false);

  useEffect(() => {

      forceNew: true,
      reconnection: true,

      setReconnectAttempts(0);

    socketInstance.on('disconnect', (reason) => {
      setIsConnected(false);
      console.log('WebSocket disconnected:', reason);
      
      if (reason === 'io server disconnect') {
        // 服务器断开连接，手动重连
        socketInstance.connect();
      }
    });

    socketInstance.on('reconnect_attempt', (attempt) => {
      setReconnectAttempts(attempt);
      console.log(`Reconnection attempt ${attempt}/${maxReconnectAttempts}`);
    });

    socketInstance.on('reconnect_failed', () => {
      console.error('WebSocket reconnection failed');
      // 可以显示错误提示给用户
    });

    Object.entries(options).forEach(([event, handler]) => {

      }
    });

    setSocket(socketInstance);

      socketInstance.disconnect();
    };
  }, [url]);
```

```typescript
// 消息合并和批处理
export class MessageOptimizer {
  private messageQueue: DiscussionMessage[] = [];
  private batchTimer: NodeJS.Timeout | null = null;
  private readonly BATCH_SIZE = 10;
  private readonly BATCH_DELAY = 100; // ms

  addMessage(message: DiscussionMessage, callback: (messages: DiscussionMessage[]) => void) {
    this.messageQueue.push(message);

    if (this.messageQueue.length >= this.BATCH_SIZE) {
      this.flushQueue(callback);
    } else if (!this.batchTimer) {
      this.batchTimer = setTimeout(() => {
        this.flushQueue(callback);
      }, this.BATCH_DELAY);
    }
  }

  private flushQueue(callback: (messages: DiscussionMessage[]) => void) {
    if (this.messageQueue.length > 0) {
      const messages = [...this.messageQueue];
      this.messageQueue = [];
      callback(messages);
    }

    if (this.batchTimer) {
      clearTimeout(this.batchTimer);
      this.batchTimer = null;
    }
  }
}
```
