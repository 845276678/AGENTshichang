# 创意提交页面设计规范文档

## 1. 页面概述

### 1.1 页面职责
创意提交页面是用户与AI Agent市场交互的关键入口，负责：
- 引导用户表达创意想法
- 提供多种输入方式（文字、语音、图片）
- 实时预估创意价值和Agent兴趣度
- 优化提交体验，提高转化率

### 1.2 目标用户
- **新手用户**：需要引导和模板帮助
- **经验用户**：快速提交，高级功能
- **移动用户**：语音输入优先

### 1.3 关键指标
- 提交完成率 > 85%
- 平均停留时间 > 3分钟
- 语音输入使用率 > 30%
- 用户满意度 > 4.2/5

// components/submit/IdeaSubmissionForm.tsx
import { useState, useRef, useCallback } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { VoiceInput } from './VoiceInput';
import { ImageUpload } from './ImageUpload';
import { TextEditor } from './TextEditor';
import { TagSelector } from './TagSelector';
import { SubmissionActions } from './SubmissionActions';

interface IdeaSubmissionFormProps {
  data: SubmissionData;
  onChange: (data: SubmissionData) => void;
  onSubmit: (data: SubmissionData) => Promise<void>;
}

export const IdeaSubmissionForm: React.FC<IdeaSubmissionFormProps> = ({
  data,
  onChange,
  onSubmit
}) => {
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [validationErrors, setValidationErrors] = useState<ValidationErrors>({});
  const formRef = useRef<HTMLFormElement>(null);

  // 表单验证
  const validateForm = useCallback((data: SubmissionData): ValidationErrors => {
    const errors: ValidationErrors = {};
    
    if (!data.content || data.content.trim().length < 10) {
      errors.content = '创意描述至少需要10个字符';
    }
    
    if (data.content && data.content.length > 1000) {
      errors.content = '创意描述不能超过1000个字符';
    }
    
    if (!data.category) {
      errors.category = '请选择创意类别';
    }
    
    if (data.tags.length === 0) {
      errors.tags = '请至少添加一个标签';
    }
    
    return errors;
  }, []);

  // 提交处理
  const handleSubmit = useCallback(async (e: React.FormEvent) => {
    e.preventDefault();
    
    const errors = validateForm(data);
    setValidationErrors(errors);
    
    if (Object.keys(errors).length > 0) {
      // 滚动到第一个错误
      const firstErrorElement = formRef.current?.querySelector('[data-error="true"]');
      firstErrorElement?.scrollIntoView({ behavior: 'smooth', block: 'center' });
      return;
    }
    
    setIsSubmitting(true);
    try {
      await onSubmit(data);
    } catch (error) {
      console.error('Submission error:', error);
    } finally {
      setIsSubmitting(false);
    }
  }, [data, validateForm, onSubmit]);

  return (
    <motion.div
      initial={{ opacity: 0, y: 20 }}
      animate={{ opacity: 1, y: 0 }}
      transition={{ duration: 0.5 }}
      className="bg-white rounded-lg shadow-sm border p-6"
    >
      <div className="mb-6">
        <h2 className="text-2xl font-bold text-gray-900 mb-2">
          分享你的创意
        </h2>
        <p className="text-gray-600">
          详细描述你的想法，让AI专家们为你的创意估价和改造
        </p>
      </div>

      <form ref={formRef} onSubmit={handleSubmit} className="space-y-6">
        {/* 输入模式选择 */}
        <InputModeSelector
          mode={data.inputMode}
          onChange={(mode) => onChange({ ...data, inputMode: mode })}
        />

        {/* 动态输入区域 */}
        <AnimatePresence mode="wait">
          {data.inputMode === 'text' && (
            <motion.div
              key="text-input"
              initial={{ opacity: 0, x: -20 }}
              animate={{ opacity: 1, x: 0 }}
              exit={{ opacity: 0, x: 20 }}
              transition={{ duration: 0.3 }}
            >
              <TextEditor
                value={data.content}
                onChange={(content) => onChange({ ...data, content })}
                error={validationErrors.content}
                placeholder="描述你的创意想法..."
                maxLength={1000}
                showWordCount={true}
              />
            </motion.div>
          )}

          {data.inputMode === 'voice' && (
            <motion.div
              key="voice-input"
              initial={{ opacity: 0, x: -20 }}
              animate={{ opacity: 1, x: 0 }}
              exit={{ opacity: 0, x: 20 }}
              transition={{ duration: 0.3 }}
            >
              <VoiceInput
                onTranscript={(transcript) => onChange({ ...data, content: transcript })}
                error={validationErrors.content}
                language="zh-CN"
                maxDuration={300} // 5分钟
              />
            </motion.div>
          )}

          {data.inputMode === 'image' && (
            <motion.div
              key="image-input"
              initial={{ opacity: 0, x: -20 }}
              animate={{ opacity: 1, x: 0 }}
              exit={{ opacity: 0, x: 20 }}
              transition={{ duration: 0.3 }}
            >
              <ImageUpload
                onImageAnalysis={(content) => onChange({ ...data, content })}
                onImagesChange={(attachments) => onChange({ ...data, attachments })}
                error={validationErrors.content}
                maxFiles={5}
                maxSize={10 * 1024 * 1024} // 10MB
              />
            </motion.div>
          )}
        </AnimatePresence>

        {/* 类别选择 */}
        <CategorySelector
          value={data.category}
          onChange={(category) => onChange({ ...data, category })}
          error={validationErrors.category}
        />

        {/* 标签选择 */}
        <TagSelector
          tags={data.tags}
          onChange={(tags) => onChange({ ...data, tags })}
          error={validationErrors.tags}
          suggestions={getTagSuggestions(data.content, data.category)}
        />

        {/* 高级选项 */}
        <AdvancedOptions
          data={data}
          onChange={onChange}
        />

        {/* 提交按钮 */}
        <SubmissionActions
          isValid={Object.keys(validateForm(data)).length === 0}
          isSubmitting={isSubmitting}
          onSubmit={handleSubmit}
          onSaveDraft={() => saveDraft(data)}
        />
      </form>
    </motion.div>
  );
};

#### 2.2.2 语音输入组件

```tsx
// components/submit/VoiceInput.tsx
import { useState, useRef, useCallback, useEffect } from 'react';
import { motion } from 'framer-motion';
import { useSpeechRecognition } from '@/hooks/useSpeechRecognition';

interface VoiceInputProps {
  onTranscript: (transcript: string) => void;
  error?: string;
  language?: string;
  maxDuration?: number;
}

export const VoiceInput: React.FC<VoiceInputProps> = ({
  onTranscript,
  error,
  language = 'zh-CN',
  maxDuration = 300
}) => {
  const [isRecording, setIsRecording] = useState(false);
  const [transcript, setTranscript] = useState('');
  const [duration, setDuration] = useState(0);
  const intervalRef = useRef<NodeJS.Timeout>();

  const {
    isSupported,
    isListening,
    start,
    stop,
    transcript: speechTranscript
  } = useSpeechRecognition({
    continuous: true,
    interimResults: true,
    language
  });

  // 开始录音
  const startRecording = useCallback(async () => {
    if (!isSupported) {
      alert('您的浏览器不支持语音输入功能');
      return;
    }

    try {
      setIsRecording(true);
      setDuration(0);
      
      // 启动语音识别
      await start();
      
      // 开始计时
      intervalRef.current = setInterval(() => {
        setDuration(prev => {
          if (prev >= maxDuration) {
            stopRecording();
            return prev;
          }
          return prev + 1;
        });
      }, 1000);
    } catch (error) {
      console.error('Failed to start recording:', error);
      setIsRecording(false);
    }
  }, [isSupported, start, maxDuration]);

  // 停止录音
  const stopRecording = useCallback(() => {
    setIsRecording(false);
    stop();
    
    if (intervalRef.current) {
      clearInterval(intervalRef.current);
    }
    
    // 将最终转录文本传递给父组件
    if (transcript.trim()) {
      onTranscript(transcript.trim());
    }
  }, [stop, transcript, onTranscript]);

  // 更新转录文本
  useEffect(() => {
    if (speechTranscript) {
      setTranscript(speechTranscript);
    }
  }, [speechTranscript]);

  // 格式化时间显示
  const formatTime = (seconds: number) => {
    const mins = Math.floor(seconds / 60);
    const secs = seconds % 60;
    return `${mins}:${secs.toString().padStart(2, '0')}`;
  };

  return (
    <div className="voice-input-container">
      {/* 录音界面 */}
      <div className={`
        relative bg-gradient-to-br from-blue-50 to-purple-50 
        rounded-2xl p-8 border-2 transition-all duration-300
        ${isRecording ? 'border-red-400 shadow-lg' : 'border-gray-200'}
        ${error ? 'border-red-400' : ''}
      `}>
        {/* 录音可视化 */}
        <div className="flex flex-col items-center space-y-6">
          {/* 麦克风按钮 */}
          <motion.button
            type="button"
            onClick={isRecording ? stopRecording : startRecording}
            className={`
              w-20 h-20 rounded-full flex items-center justify-center
              transition-all duration-300 border-4
              ${isRecording 
                ? 'bg-red-500 border-red-300 shadow-lg scale-110' 
                : 'bg-blue-500 border-blue-300 hover:bg-blue-600 hover:scale-105'
              }
            `}
            whileTap={{ scale: 0.95 }}
          >
            {isRecording ? (
              <StopIcon className="w-8 h-8 text-white" />
            ) : (
              <MicrophoneIcon className="w-8 h-8 text-white" />
            )}
          </motion.button>

          {/* 录音状态指示 */}
          {isRecording && (
            <motion.div
              initial={{ opacity: 0, scale: 0.8 }}
              animate={{ opacity: 1, scale: 1 }}
              className="flex items-center space-x-4"
            >
              <div className="flex space-x-1">
                {[...Array(5)].map((_, i) => (
                  <motion.div
                    key={i}
                    className="w-1 bg-red-500 rounded-full"
                    animate={{
                      height: [4, 16, 4],
                    }}
                    transition={{
                      repeat: Infinity,
                      duration: 0.8,
                      delay: i * 0.1,
                    }}
                  />
                ))}
              </div>
              <span className="text-red-600 font-medium">
                正在录音 {formatTime(duration)}
              </span>
            </motion.div>
          )}

          {/* 操作提示 */}
          <div className="text-center">
            <p className="text-gray-600 mb-2">
              {isRecording ? '说出你的创意想法' : '点击麦克风开始语音输入'}
            </p>
            <p className="text-sm text-gray-500">
              最长录音时间：{Math.floor(maxDuration / 60)}分钟
            </p>
          </div>

          {/* 快捷操作 */}
          {isRecording && (
            <div className="flex space-x-4">
              <button
                type="button"
                onClick={stopRecording}
                className="px-4 py-2 bg-gray-200 text-gray-700 rounded-lg hover:bg-gray-300 transition-colors"
              >
                完成录音
              </button>
              <button
                type="button"
                onClick={() => {
                  setTranscript('');
                  startRecording();
                }}
                className="px-4 py-2 bg-yellow-200 text-yellow-800 rounded-lg hover:bg-yellow-300 transition-colors"
              >
                重新录音
              </button>
            </div>
          )}
        </div>
      </div>

      {/* 转录文本预览 */}
      {transcript && (
        <motion.div
          initial={{ opacity: 0, y: 20 }}
          animate={{ opacity: 1, y: 0 }}
          className="mt-6 p-4 bg-white rounded-lg border"
        >
          <h4 className="font-medium text-gray-900 mb-2">语音转录结果：</h4>
          <p className="text-gray-700 leading-relaxed">{transcript}</p>
          
          <div className="mt-4 flex space-x-3">
            <button
              type="button"
              onClick={() => onTranscript(transcript)}
              className="px-4 py-2 bg-green-500 text-white rounded-lg hover:bg-green-600 transition-colors"
            >
              使用此文本
            </button>
            <button
              type="button"
              onClick={() => setTranscript('')}
              className="px-4 py-2 bg-gray-200 text-gray-700 rounded-lg hover:bg-gray-300 transition-colors"
            >
              清除重录
            </button>
          </div>
        </motion.div>
      )}

      {/* 错误提示 */}
      {error && (
        <motion.div
          initial={{ opacity: 0 }}
          animate={{ opacity: 1 }}
          className="mt-2 text-red-600 text-sm"
        >
          {error}
        </motion.div>
      )}
    </div>
  );
};
```

export interface SubmitIdeaRequest {
  content: string;
  category: string;
  tags: string[];
  privacy: 'public' | 'private';
  allowDiscussion: boolean;
  attachments?: {
    filename: string;
    contentType: string;
    size: number;
    data: string; // base64 encoded
  }[];
  expectedValue?: number;
  deadline?: string;
}

export interface SubmitIdeaResponse {
  success: boolean;
  data?: {
    ideaId: string;
    estimatedDiscussionStart: string;
    queuePosition: number;
  };
  error?: string;
}

// 实现示例
export async function POST(request: Request) {
  try {
    const session = await getSession(request);
    if (!session) {
      return NextResponse.json(
        { success: false, error: 'Unauthorized' },
        { status: 401 }
      );
    }

    const body: SubmitIdeaRequest = await request.json();
    
    // 验证用户配额
    const quota = await checkUserQuota(session.user.id);
    if (!quota.canSubmit) {
      return NextResponse.json(
        { success: false, error: 'Daily submission limit exceeded' },
        { status: 429 }
      );
    }

    // 内容审核
    const moderationResult = await moderateContent(body.content);
    if (!moderationResult.approved) {
      return NextResponse.json(
        { success: false, error: 'Content rejected by moderation' },
        { status: 400 }
      );
    }

    // 处理附件上传
    const attachmentUrls = await processAttachments(body.attachments);

    // 创建创意记录
    const idea = await prisma.idea.create({
      data: {
        userId: session.user.id,
        content: body.content,
        category: body.category,
        tags: body.tags,
        privacy: body.privacy,
        allowDiscussion: body.allowDiscussion,
        attachments: attachmentUrls,
        status: 'pending',
        expectedValue: body.expectedValue,
        deadline: body.deadline ? new Date(body.deadline) : null
      }
    });

    // 触发AI Agent评估流程
    await triggerAgentEvaluation(idea.id);

    // 计算队列位置
    const queuePosition = await getDiscussionQueuePosition(idea.id);

    return NextResponse.json({
      success: true,
      data: {
        ideaId: idea.id,
        estimatedDiscussionStart: calculateEstimatedStart(queuePosition),
        queuePosition
      }
    });
  } catch (error) {
    console.error('Submit idea error:', error);
    return NextResponse.json(
      { success: false, error: 'Internal server error' },
      { status: 500 }
    );
  }
}

// utils/fileUpload.ts
export async function processAttachments(attachments?: SubmitIdeaRequest['attachments']) {
  if (!attachments || attachments.length === 0) {
    return [];
  }

  const uploadPromises = attachments.map(async (attachment) => {
    // 验证文件类型和大小
    validateFile(attachment);
    
    // 生成安全的文件名
    const filename = generateSecureFilename(attachment.filename);
    
    // 上传到云存储
    const uploadResult = await uploadToCloudStorage({
      filename,
      contentType: attachment.contentType,
      data: Buffer.from(attachment.data, 'base64')
    });
    
    return {
      originalName: attachment.filename,
      storedName: filename,
      url: uploadResult.url,
      contentType: attachment.contentType,
      size: attachment.size
    };
  });

  return Promise.all(uploadPromises);
}

function validateFile(attachment: NonNullable<SubmitIdeaRequest['attachments']>[0]) {
  const allowedTypes = [
    'image/jpeg',
    'image/png', 
    'image/gif',
    'image/webp',
    'application/pdf',
    'text/plain'
  ];
  
  if (!allowedTypes.includes(attachment.contentType)) {
    throw new Error(`File type ${attachment.contentType} not allowed`);
  }
  
  if (attachment.size > 10 * 1024 * 1024) { // 10MB
    throw new Error('File size exceeds 10MB limit');
  }
}

// utils/contentModeration.ts
export async function moderateContent(content: string) {
  try {
    // 使用多种审核策略
    const [keywordCheck, aiCheck, spamCheck] = await Promise.all([
      checkKeywords(content),
      checkWithAI(content),
      checkSpam(content)
    ]);

    const isApproved = keywordCheck.passed && 
                     aiCheck.score < 0.8 && 
                     spamCheck.probability < 0.5;

    return {
      approved: isApproved,
      confidence: Math.min(keywordCheck.confidence, aiCheck.confidence),
      flags: [
        ...keywordCheck.flags,
        ...aiCheck.flags,
        ...spamCheck.flags
      ].filter(Boolean)
    };
  } catch (error) {
    console.error('Content moderation error:', error);
    // 审核失败时采用保守策略
    return {
      approved: false,
      confidence: 0,
      flags: ['moderation_error']
    };
  }
}

async function checkWithAI(content: string) {
  const response = await openai.createModerationCompletion({
    input: content
  });

  const result = response.data.results[0];
  
  return {
    score: Math.max(...Object.values(result.category_scores)),
    confidence: 0.9,
    flags: Object.entries(result.categories)
      .filter(([_, flagged]) => flagged)
      .map(([category, _]) => category)
  };
}

// hooks/useSpeechRecognition.ts
export const useSpeechRecognition = (options: SpeechRecognitionOptions = {}) => {
  const [isSupported, setIsSupported] = useState(false);
  const [isListening, setIsListening] = useState(false);
  const [transcript, setTranscript] = useState('');
  const recognitionRef = useRef<SpeechRecognition | null>(null);

  useEffect(() => {
    const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
    
    if (SpeechRecognition) {
      setIsSupported(true);
      
      const recognition = new SpeechRecognition();
      recognition.continuous = options.continuous || false;
      recognition.interimResults = options.interimResults || false;
      recognition.lang = options.language || 'zh-CN';

      recognition.onresult = (event) => {
        let finalTranscript = '';
        let interimTranscript = '';

        for (let i = event.resultIndex; i < event.results.length; i++) {
          const transcript = event.results[i][0].transcript;
          if (event.results[i].isFinal) {
            finalTranscript += transcript;
          } else {
            interimTranscript += transcript;
          }
        }

        setTranscript(finalTranscript || interimTranscript);
      };

      recognition.onstart = () => setIsListening(true);
      recognition.onend = () => setIsListening(false);
      recognition.onerror = (event) => {
        console.error('Speech recognition error:', event.error);
        setIsListening(false);
      };

      recognitionRef.current = recognition;
    }

    return () => {
      if (recognitionRef.current) {
        recognitionRef.current.abort();
      }
    };
  }, [options.continuous, options.interimResults, options.language]);

  const start = useCallback(async () => {
    if (recognitionRef.current && !isListening) {
      setTranscript('');
      recognitionRef.current.start();
    }
  }, [isListening]);

  const stop = useCallback(() => {
    if (recognitionRef.current && isListening) {
      recognitionRef.current.stop();
    }
  }, [isListening]);

  return {
    isSupported,
    isListening,
    transcript,
    start,
    stop
  };
};

export const useDebounce = <T>(value: T, delay: number): T => {
  const [debouncedValue, setDebouncedValue] = useState<T>(value);

  useEffect(() => {
    const handler = setTimeout(() => {
      setDebouncedValue(value);
    }, delay);

    return () => {
      clearTimeout(handler);
    };
  }, [value, delay]);

  return debouncedValue;
};
