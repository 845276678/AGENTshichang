```typescript
function analyzeIdea(ideaTitle: string, ideaDescription: string): IdeaCharacteristics {
  // 1. 关键词提取和分类
  const keywords = extractKeywords(ideaTitle + " " + ideaDescription)
  const category = classifyByKeywords(keywords)

  // 2. AI能力需求分析
  const aiNeeds = detectAICapabilities(ideaDescription)

  // 3. 技术复杂度评估
  const techComplexity = assessTechnicalComplexity(keywords, aiNeeds)

  // 4. 市场特征分析
  const marketInfo = analyzeMarketCharacteristics(category, keywords)

  return {
    category,
    aiCapabilities: aiNeeds,
    technicalComplexity: techComplexity,
    ...marketInfo
  }
}
```
